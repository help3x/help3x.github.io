フィルターの解除と再指定

// いまのフィルター条件を取得する
const similarMcGroupFilterValue = this.grid.getColumn({ dataIndx: this.dataIndexes.similarMcGroup, }).filter.value;

// 一度フィルター条件を消去する
this.grid.filter({
  oper: "replace",
  data: [],
});

// もう一度フィルター条件を指定する
if (!StringValidator.isNullOrEmpty(similarMcGroupFilterValue)) {
  this.grid.filter({
    oper: "replace",
    data: [
      {
        dataIndx: this.dataIndexes.similarMcGroup,
        condition: "equal",
        value: similarMcGroupFilterValue,
      }
    ],
  })
}
